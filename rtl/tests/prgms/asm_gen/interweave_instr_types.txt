// basically random instruction types one after the other
// throw in inter-instruction dependencies to keep things interesting
addi,  x1, x0,  1  // x1 = 1    ====
lw  ,  x2, x0,  2  // x2 = 0
add ,  x2, x1, x1  // x2 = 2    ====
sw  ,  x2, x2, -2  // DMEM[0] = 2
sub ,  x3, x1, x2  // x3 = -1
sll ,  x3, x3, x1  // x3 = -2
lw  ,  x4, x2,  0  // x4 = DMEM[2] (0)
sw  ,  x0, x3,  0  // DMEM[0] = -2
sll ,  x9, x3, x1  // x9 = -4
sra , x10, x3, x1  // x10 = -1  ====
sll ,  x9, x3, x1  // x9 = -4   ====
sw  ,  x0, x3,  1  // DMEM[1] = -2
lw  , x11, x0,  1  // x11 = DMEM[1] = -2 ====
add ,  x4, x9, x9  // x4 = -8
add ,  x5, x4, x2  // x5 = -6
sw  ,  x0, x9,  4  // DMEM[4] = -4
add ,  x6, x2, x4  // x6 = -6   ====
sub ,  x7, x5, x4  // x7 = 2
add ,  x7, x7, x7  // x7 = 4
lw  ,  x0, x0,  4  // nothing should happen (writing to x0)
lw  , x12, x0,  4  // x12 = -4  ====
sub ,  x8, x4, x7  // x8 = -12  ====
add ,  x7, x7, x7  // x7 = 8
add ,  x7, x7, x7  // x7 = 16   ====
sub ,  x3, x2, x7  // x3 = -14  ====
add ,  x4, x3, x3  // x4 = -28 ====
add ,  x5, x4, x2  // x5 = -26 ====
